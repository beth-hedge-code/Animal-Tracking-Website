Got it üëç ‚Äî right now your while loop just logs text, but you want it to actually inject cards into your HTML file dynamically. To do that, you‚Äôll need to manipulate the DOM instead of using console.log.
Here‚Äôs a clean example that takes your loop and appends a new card for each animal:
<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8">
  <title>Animal Cards</title>
  <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.0/dist/css/bootstrap.min.css" rel="stylesheet">
</head>
<body>
  <!-- Container where cards will be added -->
  <div id="card-container" class="d-flex flex-wrap justify-content-center"></div>

  <script>
    let animalNum = 5; // number of animals
    let animals = [
      { name: "Bear", img: "https://placebear.com/g/150/150", link: "imgs/bear.png" },
      { name: "Cow", img: "imgs/cow.png", link: "imgs/cow.png" },
      { name: "Dog", img: "https://place-puppy.com/150x150", link: "imgs/dog.png" },
      { name: "Cat", img: "https://placekitten.com/150/150", link: "imgs/cat.png" },
      { name: "Horse", img: "imgs/horse.png", link: "imgs/horse.png" }
    ];

    let container = document.getElementById("card-container");

    let i = 0;
    while (animalNum > 0) {
      let animal = animals[i];

      // Create card element
      let card = document.createElement("span");
      card.className = "position-relative m-3"; // spacing between cards
      card.innerHTML = `
        <div class="card" style="width: 10rem;">
          <div class="card-header bg-dark text-white text-center">${animal.name}</div>
          <img src="${animal.img}" class="card-img-center" alt="animal photo">
          <div class="card-body d-flex justify-content-center bg-warning">
            <button type="button" class="btn btn-warning">
              <a href="${animal.link}" class="text-dark text-decoration-none">${animal.name.toUpperCase()} FARM</a>
            </button>
          </div>
        </div>
      `;

      // Append card to container
      container.appendChild(card);

      // Decrease counter
      animalNum -= 1;
      i++;
    }
  </script>
</body>
</html>

üîë Key points:
- Instead of console.log, use document.createElement and innerHTML to build the card.
- Append each card to a container (#card-container).
- Use an array of animals so each card can have unique data (name, image, link).
- The while loop decreases animalNum until it hits zero, creating one card per iteration.
üëâ This way, when you open the HTML file in a browser, you‚Äôll see multiple cards generated dynamically.
